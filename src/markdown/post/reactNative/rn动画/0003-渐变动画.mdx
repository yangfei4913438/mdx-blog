> 非常简单的动画，直接看例子就能明白。

### 例子

```javascript
import React, { useRef } from 'react';
import { Animated, Button, View } from 'react-native';

const Demo = () => {
  // 值也必须通过Animated来声明
  // useRef 是为了避免数据被无辜刷新。
  const opacity = useRef(new Animated.Value(0.3)).current;

  const handlePress = () => {
    Animated.timing(opacity, {
      toValue: 1,
      duration: 1000,
      useNativeDriver: false,
    }).start();

    setTimeout(() => {
      Animated.timing(opacity, {
        toValue: 0.3,
        duration: 1000,
        useNativeDriver: false,
      }).start();
    }, 1000);
  };

  return (
    <View className='w-full h-full bg-white pt-8 px-12'>
      <Button title={'执行动画'} onPress={handlePress} />
      {/* 专用动画的view */}
      <Animated.View style={{ opacity }} className='mt-12 w-24 h-24 rounded-xl bg-blue-900' />
    </View>
  );
};

export default Demo;
```
