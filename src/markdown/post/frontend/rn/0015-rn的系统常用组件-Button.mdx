> 📌 无法定制ui的简单按钮。适用于对UI没要求的场景.

<div className='py-3'>过于简单，直接看例子吧。。。</div>

```javascript
import React from 'react';
import { Button, View } from 'react-native';

const ButtonDemo = () => {
  return (
    <View className='size-full bg-white p-12'>
      <Button
        title={'按钮'} // 按钮文字
        color={'green'} // 只能写这种纯颜色，或者`#f1f1f1`这种直接的颜色，动画效果继承系统默认的。不能使用`style`或则`class`自定义.
        disabled={false} // 是否禁用
        onPress={(e) => console.log(e.nativeEvent)} // 点击回调
      />
    </View>
  );
};

export default ButtonDemo;
```
